> 阶段：Phase 1-6 | 前置依赖：S01 | 来源：v2.0 §4.1 + §4.4 + §4.5 + §4.6 + §4.8

## 1. 概述

本模块定义前端完整实现，包括 Store 体系（5个 Store）、首页 TabBar、自选股列配置、Store 完整定义（含初始化锁与事件监听）、工具函数与交互细节。前端基于 Vue 3 + TypeScript + Pinia + Naive UI 构建。

## 4. 核心逻辑

### 4.1 Store 初始化锁

```typescript
// stores/config.ts
export const useConfigStore = defineStore('config', () => {
  const config = ref<AppConfig | null>(null);
  let _inited = false;
  let _initPromise: Promise<void> | null = null;

  async function init() {
    if (_inited) return;
    if (_initPromise) return _initPromise;

    _initPromise = (async () => {
      const saved = await invoke<AppConfig>('get_settings');
      config.value = deepMerge(DEFAULT_CONFIG, saved);
      // 补充遗漏字段
      if (config.value.showAddTextCount == null) config.value.showAddTextCount = 0;
      if (config.value.klineAdjust == null) config.value.klineAdjust = 'forward';
      if (config.value.peDisplayType == null) config.value.peDisplayType = 'ttm';  // v2.0
      _inited = true;
    })().finally(() => {
      _initPromise = null;
    });

    return _initPromise;
  }

  return { config, init };
});
```

### 4.2 数字格式化工具函数

```typescript
// lib/format.ts
export function formatPnlAmount(value: number): string {
  const abs = Math.abs(value);
  if (abs > 10000) return (value / 10000).toFixed(1) + 'w';
  if (abs > 1000) return (value / 1000).toFixed(1) + 'k';
  return value.toFixed(0);
}

export function formatAmount(value: number): string {
  if (value > 1e8) return (value / 1e8).toFixed(2) + '亿';
  if (value > 1e4) return (value / 1e4).toFixed(2) + '万';
  return value.toFixed(2);
}

export function formatPercent(value: number): string {
  return (value >= 0 ? '+' : '') + value.toFixed(2) + '%';
}

export function formatPlateFields(f6: number, f20: number, f22: number, f8: number) {
  return {
    f6str: (f6 / 1e8).toFixed(2) + '亿',
    f20str: (f20 / 1e8).toFixed(0) + '亿',
    f22str: f22 + '%',
    f8str: f8 + '%',
  };
}
```

### 4.4 防止滚动穿透

```typescript
// composables/useScrollLock.ts
export function usePreventScrollThrough() {
  function preventScrollThrough(event: WheelEvent) {
    const drawerEl = (event.currentTarget as HTMLElement);
    const scrollContainer = findScrollContainer(event.target as HTMLElement, drawerEl);
    if (!scrollContainer) { event.preventDefault(); return; }
    const { deltaY } = event;
    const atTop = scrollContainer.scrollTop <= 0;
    const atBottom = scrollContainer.scrollTop + scrollContainer.clientHeight
      >= scrollContainer.scrollHeight - 1;
    if ((atTop && deltaY < 0) || (atBottom && deltaY > 0)) {
      event.preventDefault();
    }
  }

  function findScrollContainer(el: HTMLElement, stop: HTMLElement): HTMLElement | null {
    let current: HTMLElement | null = el;
    while (current && current !== stop) {
      if (current.scrollHeight > current.clientHeight) return current;
      current = current.parentElement;
    }
    return null;
  }

  return { preventScrollThrough };
}
```

```typescript
function getSparklineStyle(colorType: string, theme: 'light' | 'dark') {
  if (colorType === 'blue') {
    return theme === 'dark'
      ? { stroke: 'rgba(102,169,255,0.34)', fill: 'rgba(17,24,39,0.03)' }
      : { stroke: '#8eecfe', fill: '#e7fafe' };
  }
  if (colorType === 'green') {
    return { stroke: '#26a69a', fill: 'none' };
  }
  // red（默认）：涨红跌绿，由 getColor() 动态判断
  return null;
}
```
