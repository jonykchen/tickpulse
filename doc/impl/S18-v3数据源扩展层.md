## 4. 东财统一限流器

> 来源：v3.0 §2.3（源自 a-stock-data em_get()）

```rust
/// 东财统一限流器（源自 a-stock-data 的 em_get() 设计）
/// 问题：东财风控阈值实测 — 每秒>5/并发≥10/1分≥200/5分≥300 触发封禁
/// 方案：所有东财请求统一经此入口，串行限流 + 随机抖动 + 会话复用
pub struct EmRateLimiter {
    last_request: Mutex<Instant>,
    min_interval: Duration,         // 默认 1.0s
    jitter_range: (f64, f64),       // 随机抖动 0.1-0.5s
    session: reqwest::Client,       // Keep-Alive 会话复用
}

impl EmRateLimiter {
    pub fn new() -> Self {
        Self {
            last_request: Mutex::new(Instant::now() - Duration::from_secs(10)),
            min_interval: Duration::from_millis(1000),
            jitter_range: (0.1, 0.5),
            session: reqwest::Client::builder()
                .pool_max_idle_per_host(2)
                .tcp_keepalive(Duration::from_secs(30))
                .build()
                .unwrap(),
        }
    }

    /// 所有东财请求必须通过此方法
    /// 内置：串行限流（间隔≥1s）+ 随机抖动 + 会话复用
    pub async fn get(&self, url: &str) -> Result<reqwest::Response> {
        let jitter = {
            let mut rng = rand::thread_rng();
            let range = self.jitter_range.1 - self.jitter_range.0;
            self.jitter_range.0 + rng.gen::<f64>() * range
        };
        {
            let mut last = self.last_request.lock().await;
            let elapsed = last.elapsed();
            let wait = self.min_interval + Duration::from_secs_f64(jitter);
            if elapsed < wait {
                tokio::time::sleep(wait - elapsed).await;
            }
            *last = Instant::now();
        }
        Ok(self.session.get(url).send().await?)
    }
}
```

### 6.3 PythonSidecar Rust 结构体

> 来源：v3.0 §4.3

```rust
/// Python sidecar 管理器
pub struct PythonSidecar {
    process: Option<Child>,
    stdin: Option<ChildStdin>,
    stdout: Option<BufReader<ChildStdout>>,
}

impl PythonSidecar {
    /// 启动 sidecar 子进程
    pub async fn start(&mut self) -> Result<()> {
        let mut child = Command::new("python3")
            .arg(sidecar_path)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .spawn()?;

        self.stdin = Some(child.stdin.take().unwrap());
        self.stdout = Some(BufReader::new(child.stdout.take().unwrap()));
        self.process = Some(child);
        Ok(())
    }

    /// 发送指令并获取结果
    pub async fn execute(&mut self, cmd: &str, params: Value) -> Result<Value> {
        let request = json!({
            "action": cmd,
            "ticker": params.get("ticker"),
            ..params
        });
        self.stdin.as_mut().unwrap()
            .write_all(format!("{}\n", request).as_bytes())?;
        self.stdin.as_mut().unwrap().flush()?;

        let mut response_line = String::new();
        self.stdout.as_mut().unwrap().read_line(&mut response_line)?;
        Ok(serde_json::from_str(&response_line)?)
    }
}
```

## 7. Ticker 格式归一化

> 来源：v3.0 §10.5（源自 a-stock-data + astock-peg）

**问题**：v3.0 的 secid 格式为 `1.600519`（东方财富格式），但腾讯财经用 `sh600519`，mootdx 用纯数字 `600519`，巨潮用 `600519,gssh0600519`。不同数据源需要不同的格式转换，且用户输入也可能是 `SH600519`、`600519.SH`、中文股名。

**a-stock-data 的做法**：
- 统一内部表示为纯6位数字代码
- 每个数据源适配器自行添加前缀/后缀
- 支持中文名→代码的模糊查找（mootdx 构建名称映射表）

```rust
/// 🆕 Ticker 归一化工具
pub struct TickerNormalizer {
    name_to_code: HashMap<String, String>,  // 中文名→代码（mootdx缓存）
    code_to_name: HashMap<String, String>,  // 代码→中文名
}

impl TickerNormalizer {
    /// 归一化：接受多种输入格式，返回纯6位数字
    /// 支持: "600519", "SH600519", "600519.SH", "1.600519", "贵州茅台"
    pub fn normalize(&self, input: &str) -> Result<String> {
        let s = input.trim();

        // 1. 中文名查找
        if s.chars().any(|c| c > '\u{4e00}' && c < '\u{9fff}') {
            if let Some(code) = self.name_to_code.get(&s.replace(" ", "")) {
                return Ok(code.clone());
            }
            return Err(anyhow!("找不到股票 '{}'", s));
        }

        // 2. 去除前缀 SH/SZ/BJ/sh/sz/bj
        let s = s.trim_start_matches(|c: char| c.is_alphabetic());

        // 3. 去除后缀 .SH/.SZ/.BJ
        let s = s.split('.').next().unwrap_or(s);

        // 4. 去除东财前缀 "1."/"0."
        let s = if let Some(pos) = s.find('.') {
            &s[pos + 1..]
        } else { s };

        // 5. 验证6位纯数字
        if s.len() == 6 && s.chars().all(|c| c.is_ascii_digit()) {
            Ok(s.to_string())
        } else {
            Err(anyhow!("无效的股票代码: '{}'", input))
        }
    }

    /// 转换为腾讯财经格式: sh600519 / sz000858 / bj832000
    pub fn to_tencent(&self, code: &str) -> String {
        let prefix = if code.starts_with('6') || code.starts_with('9') { "sh" }
                     else if code.starts_with('8') { "bj" }
                     else { "sz" };
        format!("{}{}", prefix, code)
    }

    /// 转换为东方财富格式: 1.600519 / 0.000858
    pub fn to_eastmoney(&self, code: &str) -> String {
        let market = if code.starts_with('6') || code.starts_with('9') { "1" }
                     else { "0" };
        format!("{}.{}", market, code)
    }

    /// 转换为巨潮格式: 600519,gssh0600519
    pub fn to_cninfo(&self, code: &str) -> String {
        let org_prefix = if code.starts_with('6') { "gssh0" }
                         else if code.starts_with('8') || code.starts_with('4') { "gsbj0" }
                         else { "gssz0" };
        format!("{},{}{}", code, org_prefix, code)
    }
}
```

## 8. 数据供应商路由与降级链

> 来源：v3.0 §12.2（源自 TradingAgents-astock `interface.py`）

**问题**：v3.0 的数据采集固定走 Python sidecar，但同一个数据可能有多个供应商（如K线可以走 mootdx/东财/新浪/百度），没有降级链，一个挂了就全挂。

```rust
/// 🆕 数据供应商路由（Rust 端，编排 Python sidecar）
pub struct DataVendorRouter {
    sidecar: Arc<Mutex<PythonSidecar>>,
    /// 每个数据能力的供应商优先级列表
    vendor_priority: HashMap<String, Vec<String>>,
}

impl DataVendorRouter {
    /// 按 Tool 配置供应商优先级
    /// 示例：kline → ["mootdx", "eastmoney", "sina"]
    ///       pe_pb → ["tencent", "eastmoney"]
    ///       news → ["eastmoney", "cninfo"]
    pub fn default_priorities() -> Self {
        let mut map = HashMap::new();
        map.insert("kline".into(), vec!["mootdx".into(), "eastmoney".into(), "sina".into()]);
        map.insert("pe_pb_market_cap".into(), vec!["tencent".into(), "eastmoney".into()]);
        map.insert("realtime_quote".into(), vec!["tencent".into(), "eastmoney".into()]);
        map.insert("financial_snapshot".into(), vec!["mootdx".into(), "eastmoney".into()]);
        map.insert("f10".into(), vec!["mootdx".into()]);
        map.insert("news".into(), vec!["eastmoney".into()]);
        map.insert("dragon_tiger".into(), vec!["eastmoney".into()]);  // 独有
        map.insert("lockup".into(), vec!["eastmoney".into()]);        // 独有
        map.insert("margin".into(), vec!["eastmoney"].into());        // 独有
        map.insert("block_trade".into(), vec!["eastmoney".into()]);   // 独有
        map.insert("shareholder".into(), vec!["eastmoney".into()]);   // 独有
        map.insert("consensus_eps".into(), vec!["ths".into()]);       // 独有
        map.insert("hot_stocks".into(), vec!["ths".into()]);          // 独有
        map.insert("northbound".into(), vec!["ths".into()]);          // 独有
        map.insert("concept_blocks".into(), vec!["eastmoney".into()]); // 独有
        map.insert("fund_flow".into(), vec!["eastmoney".into()]);     // 独有
        map.insert("industry_ranking".into(), vec!["eastmoney".into()]); // 独有
        map.insert("announcements".into(), vec!["cninfo".into(), "mootdx".into()]);
        map.insert("income_statement".into(), vec!["sina".into()]);
        Self { sidecar: Default::default(), vendor_priority: map }
    }

    /// 按降级链依次尝试
    pub async fn fetch(&self, capability: &str, params: &Value) -> Result<Value> {
        let vendors = self.vendor_priority.get(capability)
            .ok_or_else(|| anyhow!("未注册的数据能力: {}", capability))?;

        let mut last_err = None;
        for vendor in vendors {
            match self.sidecar.execute(capability, params, vendor).await {
                Ok(data) if !data.is_null() && !data.get("error").is_some() => return Ok(data),
                Ok(data) => {
                    // 返回了但数据为空/有错误，尝试下一个供应商
                    last_err = Some(data.get("error").and_then(|e| e.as_str()).unwrap_or("空数据").to_string());
                    tracing::warn!("供应商 {} 返回空数据: {}", vendor, last_err.as_ref().unwrap());
                    continue;
                }
                Err(e) => {
                    last_err = Some(e.to_string());
                    tracing::warn!("供应商 {} 失败: {}", vendor, e);
                    continue;
                }
            }
        }

        Err(anyhow!("所有供应商均失败: {}", last_err.unwrap_or_default()))
    }
}
```

## 9. 数据层已知坑点清单

> 来源：v3.0 §12.3（源自 a-stock-data SKILL.md 实测踩坑）

> a-stock-data 经过3个主版本迭代，踩了大量数据源坑。这些坑如果不提前知道，开发时会反复踩。

| 坑点 | 详情 | 规避方案 |
|------|------|---------|
| **腾讯财经字段索引43不是PB** | 网上大量教程把索引43写成PB，实测是**振幅%**。PB在索引46 | 字段映射表必须实测验证，不抄教程 |
| **腾讯财经编码GBK** | 响应是GBK编码，UTF-8解码会乱码 | 必须用 `iconv.decode(buffer, "gbk")` |
| **东财北向数据2024-08后断供** | 东财全系北向数据净买额字段返回 NaN/0，属上游断供 | 北向资金**改用同花顺 hsgtApi** + 本地CSV自缓存 |
| **百度PAE概念板块已失效** | `getrelatedblock` 返回 `ResultCode 10003` + 空数组 | 改用东财 `slist`（`spt=3`），一次拿全 |
| **巨潮公告orgId不能硬编码** | `gssx0{code}` 导致大量601xxx股票查不到公告（平安/工行/中石油） | 动态查 `szse_stock.json` 映射表（6198只股，模块级缓存） |
| **同花顺热点加了反爬** | 部分接口返回401 | 行业对比改用东财 `clist`（零鉴权） |
| **东财个股新闻结构变更** | `cmsArticleWebOld` 直接是列表，不是 `{list:[...]}` 嵌套 | 直接遍历 `cmsArticleWebOld` 列表 |
| **新浪财报结构不直观** | `result.data.report_list`（按报告期为键的dict），不是 `result.data.{lrb}` | 遍历 `report_list` 期次，从每期 `data` 按 `item_title` 提取 |
| **push2资金流单位是元** | 不是万元，不注意换算会差1万倍 | 明确标注单位，展示时统一换算 |
| **东财非交易日返回空** | 龙虎榜/解禁等在非交易日查询返回空数组 | 非交易日跳过相关请求，前端展示"休市" |
| **大陆住宅IP被东财间歇风控** | 部分住宅IP会间歇返回 `HTTP 000`/空，非代码Bug | 加重试（3次）+ 换网络提示 |
| **百度K线自带MA** | 百度股市通K线返回时自带 `ma5/ma10/ma20`，无需本地计算 | 可作为MA数据源，减少计算量 |

**落地方案**：在 Python sidecar 的 `datafeeds.py` 中内嵌这些坑点注释，每个数据采集函数头部标注已知限制。Rust 端 `DataVendorRouter` 的降级逻辑也要覆盖这些场景。

## 10. 北向数据断供本地缓存

> 来源：v3.0 §12.4（源自 a-stock-data `hsgt_realtime` + CSV缓存）

**问题**：v3.0 只用东财API拉北向资金，但东财北向数据自2024-08后净买额字段返回NaN/0（上游断供），这个坑不处理会导致北向资金面板永远显示0。

**a-stock-data 的做法**：
- 实时分钟流向改用**同花顺 hsgtApi**（`data.hexin.cn`，262个时间点，零鉴权）
- 历史日级数据用**本地CSV自缓存**：每次拉实时数据后，收盘时自动写入本地CSV
- 历史越跑越丰富，不需要外部历史API

```rust
/// 🆕 北向资金本地缓存管理器
pub struct NorthboundCache {
    db: Arc<DbPool>,
}

impl NorthboundCache {
    /// 写入收盘快照（每日15:00后自动调用）
    pub async fn save_daily_snapshot(
        &self,
        date: NaiveDate,
        hgt_yi: Decimal,   // 沪股通累计净买入(亿元)
        sgt_yi: Decimal,   // 深股通累计净买入(亿元)
    ) -> Result<()> {
        sqlx::query(
            "INSERT OR REPLACE INTO northbound_cache (date, hgt_yi, sgt_yi, updated_at) \
             VALUES (?, ?, ?, ?)"
        )
        .bind(date.format("%Y-%m-%d").to_string())
        .bind(hgt_yi)
        .bind(sgt_yi)
        .bind(Utc::now().timestamp())
        .execute(&*self.db).await?;
        Ok(())
    }

    /// 读取最近N天历史
    pub async fn load_history(&self, n: u32) -> Result<Vec<NorthboundDaily>> {
        sqlx::query_as!(
            NorthboundDaily,
            "SELECT date, hgt_yi, sgt_yi FROM northbound_cache \
             ORDER BY date DESC LIMIT ?",
            n
        )
        .fetch_all(&*self.db).await
        .map_err(Into::into)
    }
}
```

**新增表**：
```sql
CREATE TABLE northbound_cache (
    date TEXT PRIMARY KEY,
    hgt_yi TEXT NOT NULL,       -- 沪股通累计净买入(亿元)
    sgt_yi TEXT NOT NULL,       -- 深股通累计净买入(亿元)
    updated_at INTEGER NOT NULL
);
```

## 12. ST股龙虎榜规则

> 来源：v3.0 §12.9（源自 a-stock-data SKILL.md）

**问题**：v3.0 的龙虎榜和涨跌停规则是分开的，但 A 股有一个隐含规则：**ST 股5%涨跌停更容易触发龙虎榜**（"连续三日偏离值累计达12%"），科创板20%涨跌停则较少触发。这个规则直接影响游资追踪的准确性。

```rust
pub const DRAGON_TIGER_RULES: &str = r#"
龙虎榜触发规则注意：
- ST股±5%涨跌停更容易触发龙虎榜（"连续三日偏离值累计达12%"），所以ST股上榜不代表异常
- 科创板/创业板±20%涨跌停较少触发龙虎榜（偏离值阈值高），如果上榜反而说明波动更大
- 解读龙虎榜时需要结合板块涨跌幅判断上榜是否正常
"#;
```
